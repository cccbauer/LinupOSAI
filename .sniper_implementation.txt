1. Helper method to compute safety levels:
def _compute_safety_levels(self):
    """Compute safety level for each number (how many groups hit it)"""
    levels = {}
    for num in range(0, 37):
        count = sum(1 for g in self.grupos_activos if num in GRUPOS_MAESTROS[g])
        levels[num] = count
    return levels
